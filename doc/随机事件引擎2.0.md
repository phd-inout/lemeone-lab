// 公司发展阶段定义
enum CompanyStage {
 SEED = 'SEED',       // 种子期：找方向，活下去
 MVP = 'MVP',         // 原型期：打磨产品，初见用户
 PMF = 'PMF',         // 市场契合期：用户增长，验证模式
 SCALE = 'SCALE',     // 扩张期：融资，团队扩张
 IPO = 'IPO',          // 上市期：合规，资本博弈
 TITAN = 'TITAN'       // 巨头期：全球垄断，定义生态，对抗熵增
}


// 创始人 6 维能力值接口
interface FounderAttributes {
 mkt: number; // 营销 (Marketing)
 tec: number; // 技术 (Technical)
 lrn: number; // 学习 (Learning)
 fin: number; // 财务 (Financial)
 ops: number; // 运营 (Operations)
 chr: number; // 魅力 (Charisma)
}


// 随机事件定义
interface RandomEvent {
 id: string;
 title: string;
 description: string;
 baseWeight: number; // 基础权重 (1-100)
 type: 'POSITIVE' | 'NEGATIVE' | 'NEUTRAL';
  // 最小开启阶段
 minStage: CompanyStage;


 // 能力影响因子 (负数代表能力越高概率越低)
 impactFactors: Partial<Record<keyof FounderAttributes, number>>;


 // 阶段修正系数
 stageModifiers: Partial<Record<CompanyStage, number>>;
}


class EventWeightEngine {
 /**
  * 计算事件在当前语境下的最终触发权重
  */
 public calculateFinalWeight(
   event: RandomEvent,
   founder: FounderAttributes,
   currentStage: CompanyStage
 ): number {
   // 1. 检查阶段准入
   if (this.getStageIndex(currentStage) < this.getStageIndex(event.minStage)) {
     return 0;
   }


   let multiplier = 1.0;


   // 2. 能力值修正
   for (const [attr, factor] of Object.entries(event.impactFactors)) {
     const attributeValue = founder[attr as keyof FounderAttributes];
     const attributeEffect = (attributeValue / 100) * (factor as number);
     multiplier += attributeEffect;
   }


   // 3. 阶段环境修正
   const stageModifier = event.stageModifiers[currentStage] || 1.0;
   multiplier *= stageModifier;


   // 4. 最终计算
   const finalWeight = Math.max(event.baseWeight * multiplier, event.baseWeight * 0.1);
   return Math.round(finalWeight);
 }


 private getStageIndex(stage: CompanyStage): number {
   const stages = [
     CompanyStage.SEED,
     CompanyStage.MVP,
     CompanyStage.PMF,
     CompanyStage.SCALE,
     CompanyStage.IPO,
     CompanyStage.TITAN
   ];
   return stages.indexOf(stage);
 }


 /**
  * 权重随机抽样
  */
 public selectEvent(events: RandomEvent[], founder: FounderAttributes, currentStage: CompanyStage): RandomEvent {
   const weightedEvents = events
     .map(e => ({
       ...e,
       currentWeight: this.calculateFinalWeight(e, founder, currentStage)
     }))
     .filter(e => e.currentWeight > 0);


   const totalWeight = weightedEvents.reduce((sum, e) => sum + e.currentWeight, 0);
   let random = Math.random() * totalWeight;


   for (const event of weightedEvents) {
     if (random < event.currentWeight) return event;
     random -= event.currentWeight;
   }


   return weightedEvents[0];
 }
}
